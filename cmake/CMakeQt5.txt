
find_package(PkgConfig)
set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_OSX_ARCHITECTURES x86_64)
find_package (Qt5 COMPONENTS Widgets Xml Svg Multimedia Network REQUIRED)
include_directories(
    ${Qt5Widgets_INCLUDE_DIRS}
    ${Qt5Xml_INCLUDE_DIRS}
    ${Qt5Svg_INCLUDE_DIRS}

    ${Qt5Network_INCLUDE_DIRS})

qt5_add_resources(QTERM_RCC ${QTERM_RCC_SRC})
#qt5_add_translation(QTERM_QM ${QTERM_TS})

if (WIN32)

    ${Qt5Svg_LIBRARIES}

    ${Qt5Multimedia_LIBRARIES}
    ${Qt5Network_LIBRARIES})

    set (BUILD_PROJECT_LIBRARIES
    ${Qt5Widgets_LIBRARIES}
    ${Qt5Xml_LIBRARIES}
    ${Qt5Svg_LIBRARIES}
    ${Qt5Multimedia_LIBRARIES}
    ${Qt5Network_LIBRARIES})
endif (WIN32)

    ${Qt5Widgets_DEFINITIONS}
    ${Qt5Xml_DEFINITIONS}
    ${Qt5Svg_DEFINITIONS}
    ${Qt5Multimedia_DEFINITIONS}
    ${Qt5Network_DEFINITIONS})
