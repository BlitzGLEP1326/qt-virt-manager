
set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_OSX_ARCHITECTURES x86_64)
find_package (Qt5 COMPONENTS Core Gui Xml Svg REQUIRED)
include_directories(${Qt5Widgets_INCLUDE_DIRS} ${Qt5Xml_INCLUDE_DIRS} ${Qt5Svg_INCLUDE_DIRS})
qt5_wrap_cpp (MOC_SOURCES ${HEADERS})
qt5_wrap_ui(UI_SRCS src/qtermwidget/lib/SearchBar.ui)

if (WIN32)
    set (BUILD_PROJECT_LIBRARIES ${Qt5Widgets_LIBRARIES} ${Qt5Xml_LIBRARIES} ${Qt5Svg_LIBRARIES} ${Qt5Core_QTMAIN_LIBRARIES})
else (WIN32)
    set (BUILD_PROJECT_LIBRARIES ${Qt5Widgets_LIBRARIES} ${Qt5Xml_LIBRARIES} ${Qt5Svg_LIBRARIES})
endif (WIN32)
add_definitions(${Qt5Widgets_DEFINITIONS} ${Qt5Xml_DEFINITIONS} ${Qt5Svg_DEFINITIONS})
set (QT5_BUILD 1)
